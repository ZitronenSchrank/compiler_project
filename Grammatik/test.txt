def add(s1, s2) begin:
    var sum := s1;

    while(s2) begin:
        succ(sum);
        pred(s2);
    end

    return sum;
end

def sub(s1, s2) begin:
    var retVal := s1;

    while(s2) begin:
        pred(retVal);
        pred(s2);
    end

    return retVal;
end

var r0 := 0;
var result := 0;

while(r0) begin:
    var sum := add(3, r0);
    var diff := sum(42, sum);

    loop(diff) begin:
        result := add(3, diff);
        succ(result);
    end
end

write(result);
