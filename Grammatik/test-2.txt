// Subtrahiert etwas
sub(s1, s2) begin:
    var retVal := s1;

    while(s2) begin:
        pred(retVal);
        pred(s2);
    end

    return retVal;
end

// Multipliziert etwas
mutliply(s1, s2) begin:
    var retVal := s1;

    pred(s2);
    while(s2) begin:
        retVal := add(s1, retVal);
        pred(s2);
    end

    return retVal;
end
